# problem 3 - longest substring without repeat chars

```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        
        // loop over string

        string high_str = "";
        string curr_str = "";

        int l = 0;
        int hl = 0;

        for (char c : s){
            cout << "c: " << c << "\n";
            if (curr_str.find(c)){
                l = 0;
                cout << curr_str << "\n";
                curr_str="";
            } else {
                l += 1;
                curr_str = curr_str+c;
                cout << curr_str << "\n";

                if (l > hl){
                    hl = l;
                }

                if (curr_str.size() > high_str.size()){
                    high_str = curr_str;
                }
            }
        }

        return hl;
    }
};
```
